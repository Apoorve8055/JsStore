#two table join

 FromLogic = {
                    Table1: {
                        Table: 'Receipt',
                        Column: 'ReceiptId',
                        Where: {
                            ReceiptId: [18, 17]
                        }
                    },
                    Join: 'inner',
                    Table2: {
                        Table: 'Bill',
                        Column: 'ReceiptId',
                    }
                };

Db.DbConnection.select({
                From: FromLogic,
            }, function (values) {
                if (values.length > 0) {
                    DialogBox.alert('Successfully added');
                    console.log(values);
                } else {
                    DialogBox.alert('Error Occured');
                }
            }, function (error) {
                console.log(error);
                DialogBox.alert('Error Occured');
            })
